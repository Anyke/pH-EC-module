ЕС:


uint8_t buff[24];

memset(buff, 0, 24);

Wire.begin(4, 16, 80000); // инициализация I2C

Wire.requestFrom(0x34, 23); // 0x34 - адрес, 24 - кол-во байт

Wire.readBytes(buff, 23);

Wire.endTransmission();

if (buff[0] == 0x51 && buff[1] == 0x01) { // сигнатура EC модуля

uint16_t ec = ((uint16_t)buff[3] * 256) + (uint16_t)buff[2];

uint16_t t = ((uint16_t)buff[9] * 256) + (uint16_t)buff[8]; // температура, умноженная на 100. Значение меньше или равно 100 - ошибка чтения датчика температуры

uint16_t level = ((uint16_t)buff[11] * 256) + (uint16_t)buff[10]; // значение со входа АЦП для датчика давления, если распаян усилитель

uint16_t raw_r = (float)(((int16_t)buff[5] * 256) + (uint16_t)buff[4]); // "сырое" значение измеренного на переменном токе сопротивления, из которого рассчитывается ЕС


uint16_t ecCal[2];

uint16_t rCal[2];

ecCal[0] = ((uint16_t)buff[13] * 256) + (uint16_t)buff[12];

ecCal[1] = ((uint16_t)buff[15] * 256) + (uint16_t)buff[14];


rCal[0] = ((uint16_t)buff[17] * 256) + (uint16_t)buff[16];

rCal[1] = ((uint16_t)buff[19] * 256) + (uint16_t)buff[18];

if (t > 100)

water_temp = (float)t / 100.0f;

// ....

}



pH:

uint8_t buff[20];

memset(buff, 0, 20);


Wire.begin(4, 16, 80000); // инициализация I2C

Wire.requestFrom(0x33, 20); // 0x33 - адрес, 24 - кол-во байт

Wire.readBytes(buff, 20);

Wire.endTransmission();


if (buff[0] == 0x50 && buff[1] == 0x01) { // сигнатура pH модуля

int16_t u = ((int16_t)buff[5] * 256) + (uint16_t)buff[4]; // напряжение в мВ, умноженное на 10

u /= 10;


uint16_t ph100 = ((uint16_t)buff[3] * 256) + (uint16_t)buff[2]; // расчитанный pH, умноженный на 100

float ph = (float)ph100 / 100.0f;


uint16_t phCal[2];

int16_t uCal[2];


phCal[0] = ((uint16_t)buff[9] * 256) + (uint16_t)buff[8];

phCal[1] = ((uint16_t)buff[11] * 256) + (uint16_t)buff[10];

uCal[0] = ((int16_t)buff[13] * 256) + (uint16_t)buff[12];

uCal[1] = ((int16_t)buff[15] * 256) + (uint16_t)buff[14];


int16_t raw_u = (float)(((int16_t)buff[5] * 256) + (uint16_t)buff[4]) / 10.0f; // сырое значение напряжения со входа модуля, мВ


// ....

}




Калибровка EC:


uint16_t ec1 = 992, ec2 = 2709, r1 = 821, r2 = 347;

Wire.begin(4, 16, 80000);

Wire.beginTransmission(0x34);

Wire.write(12);


Wire.write(ec1 & 0xFF);

Wire.write((ec1 / 256) & 0xFF);


Wire.write(ec2 & 0xFF);

Wire.write((ec2 / 256) & 0xFF);


Wire.write(r1 & 0xFF);

Wire.write((r1 / 256) & 0xFF);


Wire.write(r2 & 0xFF);

Wire.write((r2 / 256) & 0xFF);


Wire.write(0x00);

Wire.write(0x00);


Wire.write(0xC8); // temp coeff = 0.02 (write 200, 200d = C8h)


Wire.endTransmission(true); // false for ReSTART
